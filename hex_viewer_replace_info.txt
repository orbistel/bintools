# 1. Dosyayı görüntüle ve imzaları bul
hexview_replace.exe image.jpg 0 256 -f

# 2. JPEG imzasını '*' ile değiştir
hexview_replace.exe photo.jpg 0 0 -r "*"

# 3. PDF imzasını null byte'larıyla (0x00) değiştir
hexview_replace.exe document.pdf 0 0 -R 00

# 4. Yedek al ve değiştir
hexview_replace.exe secret.exe 0 0 -r "X" -b backup.exe

# 5. Tüm imzaları listele
hexview_replace.exe any.txt 0 0 -l

# 6. İnteraktif modda değiştir (sorar)
hexview_replace.exe data.bin 0 512 -r "#" -i

# 7. İmza ara ve hex dump göster
hexview_replace.exe unknown.dat 0 128 -f
